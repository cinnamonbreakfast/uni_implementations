bits 32

global start        

extern exit
import exit msvcrt.dll

segment data use32 class=data
    B db 11     ; 1011
    A dw 23011

segment code use32 class=code
    start:
        MOV EAX, [A]
        
        MOV BL, [B]
        AND BL, 0x0F
        
        AND AH, 0xF0
        ADD AH, BL
        
        
        push    dword 0
        call    [exit]
